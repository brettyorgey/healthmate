async function sendMsg(txt) { const inp = document.getElementById('msg'); const content = (txt || inp.value || '').trim(); if (!content) return; addMsg('user', content); inp.value = ''; document.getElementById('sendBtn').disabled = true; try { const r = await fetch('/api/mascot', { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({ message: content }) }); const data = await r.json().catch(()=>({})); if (!r.ok) { const detail = typeof data?.detail === 'string' ? ` (${data.detail})` : ''; addMsg('assistant', 'Sorry — server error' + detail); } else { addMsg('assistant', data.output || 'No response', data.sources); } } catch (e) { addMsg('assistant', 'Sorry — network error.'); } finally { document.getElementById('sendBtn').disabled = false; } }
